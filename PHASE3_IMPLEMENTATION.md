// Naver에서 수집할 데이터
const smartMoneyData = {
  ticker: "005930",  // 삼성전자

  institution: {
    buy: 1250000,      // 기관 매수량 (주)
    sell: 980000,      // 기관 매도량 (주)

    buyRate: 0.56,     // 매수 비율
    sellRate: 0.44,    // 매도 비율

    buy: 850000,

    buyRate: 0.54,
    sellRate: 0.46,
  },
  
  shortSell: {
    volume: 45000,     // 공매도 거래량 (주)

  },
  
  priceLevel: {
    high: 85000,       // 고가 영역 거래량
    mid: 120000,       // 중가 영역
